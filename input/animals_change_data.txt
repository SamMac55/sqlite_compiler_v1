// new table with no constraints on feed days
new table animal_diet {
    int diet_id(pk, not null),
    String diet_type (not null),
    int num_feed_days
};

// inserting into a table with no constraint
add animal_diet (diet_id = 1,diet_type= 'vegetarian'), (diet_id=2,diet_type='omnivore',num_feed_days=3),(diet_id=3,diet_type='carnivore',num_feed_days=5);

// show that null comparisons work
animal_diet all with num_feed_days is null;

animal_diet all with num_feed_days is not null;

// adding that information in
change animal_diet set num_feed_days=4 with diet_type is 'vegetarian';
//show it works
animal_diet all with diet_id is 1;

// remove table 
remove table animal_diet;
