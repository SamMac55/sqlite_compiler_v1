//create and populate a new table
new table song_ratings{
    int rating_id (pk),
    int user_id (not null, reference users),
    int song_id (not null, reference songs),
    double rating (not null),
    String review
};

add song_ratings (rating_id=1,user_id=1,song_id=2,rating=5,review='Amazing!'),
(rating_id=2,user_id=2,song_id=3,rating=2.5,review='Not my favorite!'),
(rating_id=3,user_id=1,song_id=1,rating=4),
(rating_id=4,user_id=2,song_id=4,rating=3.4);

//did it work?
song_ratings all;

//testing double comparison
song_ratings all with rating is 2.5;

//testing attribute comparison
song_ratings all with user_id less than song_id;

//setting multiple values in a change
change song_ratings set rating=4,review='Great song!' with rating_id is 4;

//setting a value to another attribute
change song_ratings set rating=song_id with rating_id is 3;

//did it work?
song_ratings all;

remove song_ratings with user_id is 2 and song_id at least 4;

remove song_ratings with user_id is 1 or song_id at most 2;

//show results
song_ratings all;

remove table song_ratings;
